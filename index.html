<script>

        /**

         * KOVA SONG ARCHIVE - DATA INTEGRATION

         * -------------------------------------------------------------------------

         * INSTRUCTIONS: 

         * 1. Go to your Google Sheet.

         * 2. File > Share > Publish to web.

         * 3. Select 'Entire Document' (or specific sheet) and 'Comma-separated values (.csv)'.

         * 4. Copy the URL and replace the placeholder below.

         * -------------------------------------------------------------------------

         */

        const CSV_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vT-PLACEHOLDER-ID/pub?gid=0&output=csv';


        async function fetchArchive() {

            const container = document.getElementById('archive-container');

            const loading = document.getElementById('loading-state');


            try {

                const response = await fetch(CSV_URL);

                if (!response.ok) throw new Error('Network response was not ok');

                

                const csvText = await response.text();

                const data = parseCSV(csvText);

                

                renderItems(data);

                if (loading) loading.remove();

            } catch (error) {

                console.error('Error fetching archive:', error);

                if (loading) {

                    loading.innerHTML = `<span class="text-error font-label-technical uppercase tracking-widest">Error loading data. Please check CSV URL.</span>`;

                }

            }

        }


        function parseCSV(csv) {

            const lines = csv.split(/\r?\n/);

            const result = [];

            if (lines.length === 0) return result;


            const headers = lines[0].split(',').map(h => h.trim().replace(/^"|"$/g, ''));


            for (let i = 1; i < lines.length; i++) {

                if (!lines[i].trim()) continue;

                

                // Simple CSV parsing that handles basic quoted strings

                const row = [];

                let current = '';

                let inQuotes = false;

                for (let char of lines[i]) {

                    if (char === '"') inQuotes = !inQuotes;

                    else if (char === ',' && !inQuotes) {

                        row.push(current.trim());

                        current = '';

                    } else {

                        current += char;

                    }

                }

                row.push(current.trim());


                const obj = {};

                headers.forEach((header, index) => {

                    let val = row[index] || '';

                    obj[header] = val.replace(/^"|"$/g, '');

                });

                result.push(obj);

            }

            return result;

        }


        function renderItems(items) {

            const container = document.getElementById('archive-container');

            

            items.forEach(item => {

                const isHighlighted = item.Highlight && item.Highlight.toLowerCase() === 'sí';

                const highlightClass = isHighlighted ? 'border-b-2 border-[#7ed321]' : '';

                

                const html = `

                    <div class="bg-[#ede9e1] relative flex flex-col md:flex-row items-start md:items-center justify-between p-10 border border-primary/10 ${highlightClass} gap-8">

                        <span class="absolute top-4 right-6 font-label-technical text-[10px] text-[#9a9a9a] uppercase tracking-widest opacity-80">KOVA RS / 48kHz</span>

                        <div class="flex items-center gap-10">

                            <div class="w-28 h-28 overflow-hidden shrink-0 border border-primary/5">

                                ${item.Cover ? 

                                    `<img alt="Album artwork for ${item.Song}" class="w-full h-full object-cover grayscale contrast-125 album-cover-transition hover:grayscale-0" src="${item.Cover}"/>` :

                                    `<div class="w-full h-full bg-surface-variant flex items-center justify-center">

                                        <span class="material-symbols-outlined text-5xl text-secondary opacity-50">album</span>

                                    </div>`

                                }

                            </div>

                            <div class="flex flex-col gap-2">

                                <h2 class="column-song font-headline-lg-mobile text-headline-lg-mobile text-[#0b0b0b] font-bold">${item.Song || 'Unknown Track'}</h2>

                                <p class="column-artist font-body-md text-body-md text-[#0b0b0b] font-normal">${item.Artist || 'Unknown Artist'}</p>

                                <span class="font-label-technical text-[11px] text-[#9a9a9a] uppercase tracking-widest mt-2">

                                    <span class="column-date">${item.Date || '---'}</span> | <span class="column-credits">${item.Credits || 'No metadata'}</span>

                                </span>

                            </div>

                        </div>

                        <div class="flex items-center w-full md:w-auto justify-end">

                            <a class="border border-primary bg-primary text-on-primary px-8 py-3 font-label-technical text-label-technical uppercase tracking-widest hover:bg-background hover:text-tertiary-fixed-dim hover:border-tertiary-fixed-dim transition-colors text-center" href="${item.Link || '#'}" target="_blank">

                                Escuchar

                            </a>

                        </div>

                    </div>

                `;

                container.insertAdjacentHTML('beforeend', html);

            });

        }


        // Initialize

        document.addEventListener('DOMContentLoaded', fetchArchive);

    </script>
